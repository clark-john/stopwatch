// input:
let csv = ""
let arr = [
	{time:'', number: 1 },
	{time:'', number: 2 },
	{time:'', number: 3 },
	{time:'', number: 4 },
	{time:'', number: 5 },
]

// iterate
for (let x of arr) {
	const { number, time } = x;
	csv += `${number},${time}\n`;
}

// output (csv):
// number, time

`
1,0.232
2,0.445
3,0.563
4,0.123

`
